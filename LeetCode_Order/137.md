class Solution {
   public int singleNumber(int[] nums) {
         int value = 1;
        Map<Integer, Integer> map = new HashMap();
        for (int num : nums) {
            if (map.containsKey(num)) {
                map.put(num, 3);
                continue;
            } else {
                map.put(num, 1);
            }
        }
        int result = -1;
        for (Map.Entry<Integer, Integer> entry : map.entrySet()) {
            if (value == entry.getValue()) {
                result = entry.getKey();
            }
        }
        return result;
    }
}
